```

  <HoverMenu>
    <Button>Hover me</Button>

    <template #menu>
      <HoverMenuList>
        <HoverMenuItem><IconPencilOutline />Открыть</HoverMenuItem>
        <HoverMenuItem><IconExternalLinkOutline />Открыть в новой вкладке</HoverMenuItem>
      </HoverMenuList>
    </template>
